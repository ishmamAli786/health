CREATE OR REPLACE VIEW app_view AS (SELECT S.CODE SCHEDULE_CODE,S.DOCTORS_CODE, S.TIME, S.DAYS,S.FEE,S.NO_OF_APP,H.CODE HOSPITAL_CODE, H.NAME HOSPITAL,H.AREA, H.CONTACT_NUMBER,H.EMAIL,A.CODE APP_CODE,A.PATIENTS_CODE,A.PROBLEM,A.REMARKS,A.APPOINT_DATE,A.REPORT,A.STATUS APP_STATUS FROM `hospital` H, hospital_schedule S,appointments A WHERE H.CODE = S.HOSPITALS_CODE AND A.HOSPITALS_SCHEDULE_CODE = S.CODE)
