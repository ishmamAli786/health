CREATE OR REPLACE VIEW DOCTOR_VIEW AS (SELECT D.CODE DTL_CODE, D.BACKGROUND, D.CONSULT_TYPE,D.REG_CODE,D.STATUS,S.NAME SP_NAME, S.ABBREVATION,S.BODY_PART,L.CODE DOCTOR_CODE, L.NAME, L.FATHER_NAME,L.EMAIL,L.IMG, L.CONTACT_NUMBER,L.ADDRESS FROM `doctor_detail` D, login L, specialization S WHERE D.DOCTORS_CODE = L.CODE AND D.SPECIALIZATION_CODE = S.CODE)
